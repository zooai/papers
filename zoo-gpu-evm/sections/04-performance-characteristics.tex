\section{Performance Characteristics}

\label{sec:performance}

\begin{theorem}[Linear Scaling]
\label{thm:linear}
For sequential fraction $f$ and $n$ independent transactions across $p$ SMs ($p \leq n$):
\begin{equation}
\Theta(p) = \Theta_{\text{base}} \cdot \frac{p}{1 + p \cdot f / (1 - f)}
\end{equation}
\end{theorem}

\begin{proposition}[Multi-GPU Linear Scaling]
For $k$ GPUs executing disjoint transaction subsets:
$\Theta_k = k \cdot \Theta_1 - O(k \cdot c_{\text{sync}})$
where $c_{\text{sync}}$ is per-GPU synchronization cost.
\end{proposition}
